```typescript
/**
 * Environment variable validation
 * Runs at build time and startup; throws clear error if critical vars missing
 * 
 * Critical vars for Phase 1:
 * - HOSTAWAY_API_KEY: Hostaway API authentication (server-only)
 * - PRISMIC_TOKEN: Prismic CMS API token (server-only)
 * - PUBLIC_SENTRY_DSN: Sentry error tracking (client-visible)
 * - PUBLIC_GA4_ID: Google Analytics 4 measurement ID (client-visible)
 */

interface EnvConfig {
  hostawayApiKey: string;
  pricmicToken: string;
  sentryDsn: string;
  ga4Id: string;
}

export function validateEnv(): EnvConfig {
  const env = {
    hostawayApiKey: import.meta.env.HOSTAWAY_API_KEY,
    pricmicToken: import.meta.env.PRISMIC_TOKEN,
    sentryDsn: import.meta.env.PUBLIC_SENTRY_DSN,
    ga4Id: import.meta.env.PUBLIC_GA4_ID
  };

  const missing = Object.entries(env)
    .filter(([_, value]) => !value)
    .map(([key]) => key);

  if (missing.length > 0) {
    const message =
      `\n❌ CONFIGURATION ERROR\n\n` +
      `Missing critical environment variables:\n` +
      missing.map(key => `  - ${key}`).join('\n') +
      `\n\nFor local development:\n` +
      `  1. Copy .env.example to .env.local\n` +
      `  2. Fill in all placeholder values\n` +
      `  3. Do NOT commit .env.local to git\n\n` +
      `For Cloudflare Pages production:\n` +
      `  1. Configure in Cloudflare project settings → Environment Variables\n` +
      `  2. Use PUBLIC_ prefix only for client-visible vars\n\n`;
    throw new Error(message);
  }

  return env as EnvConfig;
}

// Export as singleton for app-wide access
export const env = validateEnv();
```

```markdown
   - `PUBLIC_GA4_ID`
```

## Trust Boundaries

| Boundary | Description | Risk |
|----------|-------------|------|
| Client Browser ↔ Cloudflare CDN | Static HTML/CSS/JS delivered via HTTPS | Mitigate: HTTPS enforced by Cloudflare (automatic) |
| Local Dev ↔ API (Hostaway, Prismic) | Unencrypted API keys in `.env.local` | Mitigate: `.env.local` git-ignored; never committed |
| Astro Build ↔ Environment Variables | Secrets loaded at build time | Mitigate: `PUBLIC_` prefix enforced for client-visible vars; server-only vars never leak |
| Browser ↔ Sentry/GA4 | Page views and error data sent externally | Mitigate: PII scrubbing enabled in Sentry config (Phase 6); GA4 filtered for PII |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Disclosure | Environment variables exposed in error messages | mitigate | Catch errors with try-catch; sanitize error objects; never log raw API responses |
| T-01-02 | Disclosure | `.env.local` accidentally committed to git | mitigate | `.gitignore` includes `.env.local`; use git hooks to prevent secret commits; `.env.example` only |
| T-01-03 | Disclosure | API keys hardcoded in source code | mitigate | Use `import.meta.env` to load from environment; TypeScript linting could enforce this |
| T-01-04 | Tampering | XSS via unsanitized user input (contact form Phase 4) | mitigate | Astro auto-escapes HTML in templates; validate input server-side before processing |
| T-01-05 | Tampering | CSRF on form submissions (contact form Phase 4) | mitigate | Use POST + HTTPS (Astro default); Cloudflare CSRF protection (automatic) |
| T-01-06 | Repudiation | No audit log of who deployed which version | accept | Rationale: single-developer project; git commit log serves as audit trail; Cloudflare Pages deployment log available |
| T-01-07 | Information Disclosure | Detailed error messages leak system info | mitigate | Catch exceptions; return generic "Something went wrong" to client; log detailed errors server-side |
| T-01-08 | Denial | Build process extremely slow | mitigate | Monitor build time (Phase 1 baseline ~35-45s); Cloudflare timeout 60min (sufficient for scaling to 100+ rooms) |
